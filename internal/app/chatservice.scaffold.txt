package app

import (
	"context"
	"database/sql"
	"fmt"
	"strings"

	"changeme/internal/engine"
	"changeme/internal/store"

	llama "github.com/tcpipuk/llama-go"
	"github.com/wailsapp/wails/v3/pkg/application"
)

type ChatService struct {
	Engine *engine.Engine // set in main.go before app.New, not via a constructor
	DB     *sql.DB        // same

	app *application.App // filled in by ServiceStartup, confirmed pattern from your own scaffold
}

// ServiceStartup's signature and application.Get() are both confirmed against
// your actual generated main.go / bindings — not a guess this time.
func (s *ChatService) ServiceStartup(ctx context.Context, options application.ServiceOptions) error {
	s.app = application.Get()
	return nil
}

func (s *ChatService) SendMessage(sessionID int64, collectionID int64, prompt string) error {
	if s.app == nil {
		s.app = application.Get()
	}
	if strings.TrimSpace(prompt) == "" {
		return fmt.Errorf("prompt cannot be empty")
	}
	if s.Engine == nil || s.DB == nil {
		return fmt.Errorf("chat service not fully initialised: engine or db is nil")
	}

	queryEmb, err := s.Engine.Embed(prompt)
	if err != nil {
		return fmt.Errorf("embedding query: %w", err)
	}

	chunks, err := store.HybridSearch(s.DB, collectionID, prompt, queryEmb, 5)
	if err != nil {
		return fmt.Errorf("hybrid search: %w", err)
	}

	messages := buildMessages(chunks, prompt)

	chatCtx, err := s.Engine.ChatModel.NewContext(llama.WithContext(4096))
	if err != nil {
		return fmt.Errorf("creating chat context: %w", err)
	}
	// Deliberately no defer chatCtx.Close() at this level — SendMessage
	// returns as soon as the goroutine below is launched. Closing here would
	// free the context's underlying C++ pointer while generation is still
	// in flight — a use-after-free, not just wasted work. Close moves into
	// the goroutine, after the loop actually exits.

	deltas, errs := chatCtx.ChatStream(context.Background(), messages, llama.ChatOptions{})

	go func() {
		defer chatCtx.Close()
		for {
			select {
			case delta, ok := <-deltas:
				if !ok {
					s.emit("chat:done", sessionID)
					return
				}
				s.emit("chat:token", map[string]any{
					"sessionId": sessionID,
					"token":     delta.Content,
				})
			case err := <-errs:
				if err != nil {
					s.emit("chat:error", err.Error())
					return
				}
			}
		}
	}()

	return nil
}

func (s *ChatService) emit(name string, data ...any) {
	if s.app != nil {
		s.app.Event.Emit(name, data...)
	}
}

func buildMessages(chunks []store.ScoredChunk, prompt string) []llama.ChatMessage {
	var contextBuilder strings.Builder
	for _, chunk := range chunks {
		if chunk.Content == "" {
			continue
		}
		contextBuilder.WriteString("- ")
		contextBuilder.WriteString(chunk.Content)
		contextBuilder.WriteByte('\n')
	}

	systemPrompt := "You are a local RAG assistant. Answer from the provided context when it is relevant, and say when the context is not enough."
	if contextBuilder.Len() > 0 {
		systemPrompt += "\n\nContext:\n" + contextBuilder.String()
	}

	return []llama.ChatMessage{
		{Role: "system", Content: systemPrompt},
		{Role: "user", Content: prompt},
	}
}
